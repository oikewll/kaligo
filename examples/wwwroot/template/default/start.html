1. Use the below Go command to install Kaligo.

go get -u github.com/owner888/kaligo

2. Quick start with the example.

package main

import (
    "net/http"

    "github.com/owner888/kaligo"
)

type Index struct {
    kaligo.Controller
}

func (c *Index) Index() {
    c.JSON(http.StatusOK, kaligo.H{"message": "Hello, Kaligo!"})
}

func main() {
    r := kaligo.NewRouter()
    r.AddRoute("/", map[string]string{http.MethodGet: "Index"}, &Index{})
    kaligo.Run(r, ":80")
}
